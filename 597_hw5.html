my.chi <- function(n,df){
  n=n/2
  u1 = runif((df*n))
  u2 = runif((df*n))
  x1 = sqrt(-2*log(u1))*cos(2*pi*u2)
  x2 = sqrt(-2*log(u1))*sin(2*pi*u2)
  x = c(x1,x2)
  z = matrix(data =x , ncol = df)
  w = z**2
  w = apply(w,1,sum)
  
  return(w)
}

my.f <- function(n,a,b){
  x = my.chi(n,a)
  y = my.chi(n,b)
  f = (x/a)/(y/b)

  k=0

  y = rep(NA,n/2)
  j=0

    v2 = 2*u2 -1

    if(S<1){
      k=k+1
      x[k]=sqrt(-2*log(S)/S)*v1
      y[k]=sqrt(-2*log(S)/S)*v2
    }
  }
  #print(length(x))
  return(c(x,y))
}
